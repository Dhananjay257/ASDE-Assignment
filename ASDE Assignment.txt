Task 1: SQL


select d.Name as "DEPT_NAME",avg(s.AMT) as "AVG_MONTHLY_SALARY" from employees as e join departments as d 
on e.DEPT_ID=d.id
join salaries as s on s.EMP_ID=e.id
group by d.NAME
order by avg(s.AMT) desc limit 3;
